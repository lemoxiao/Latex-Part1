% Possible options are
%
% ltrhead     Typeset with letterhead backdrop (default).
% noltrhead   Suppress letterhead backdrop (for printing on real letterhead).
%
% block       Use block letter format (default).
% semiblock   Use semiblock letter format (indented date and signature).

%% called hmcletter.cfg in the working directory or in your
%% personal TEXMF tree, and will override the default settings in
%% the class file.  Commands defined here will override those set

% Name and address of the letter recipient.

Department of Mathematics\\
Anytown University\\
123 University Drive\\
Anytown, California  12345-6789\\
United States of America}

% [OPTIONAL] Subject line.  \subjectname can be set to define a
% header before the text in the letter, but doing so is not

% Signature.  Note two arguments.  Optional argument allows you to
% load a graphical version of your signature for use in generating
% PDFs that look like real signed letters.  (Without the optional
% argument, the class inserts blank vertical space for your
% signature.)
\signature[{\includegraphics[scale=0.25]{jrandomsig}}]%
{J.~Random User\\ Associate Professor of Mathematics}


% Date command; standard LaTeX command, so it will use the date
% you specify, fall back to
% current date if no \date command is used, or not appear at all

\date{January 1, 2009}
