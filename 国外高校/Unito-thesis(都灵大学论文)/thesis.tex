%% HEADERS AND FOOTERS
\usepackage{fancyhdr}
\pagestyle{fancy}

\rhead{} %right head
\lhead{ \leftmark} %left head
\rfoot{\thepage}

%coherce the plain stile to this (e.g. the first page of every chapter)

	\rfoot{\thepage}

}
