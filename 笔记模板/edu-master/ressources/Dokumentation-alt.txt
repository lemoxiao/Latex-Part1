% \subsection{Klasse laden und Optionen manipulieren}\label{optionen}

%
%
% \begin{decl}
%   |\sclsetup{|\meta{Option}=\meta{Wert},\dots,\meta{Option}=\meta{Wert}|}|\\
%   |\scloption{|\meta{Option}|}||{|\meta{Wert}|}|

%
% Vielen Optionen kann ein beliebiger Text (\meta{String}) zugeordnet werden. Einige Optionen erwarten eine Maßeinheit (\meta{Dim} -- \emph{Dimension}), z.\,B. in |cm|, |mm|, |pt|, |ex| oder |em|. Booleschen Optionen (\meta{Bool}) hingegen können die \emph{Wahrheitswerte} |true| (\emph{Wahr}) und |false| (\emph{Falsch}) zugeordnet werden. Boolesche Optionen können alternativ auch ohne Wert angegeben, was dem Wert |true| entspricht. D.\,h. die Angabe |twoup| entspricht |twoup=true|. Tabelle \ref{options} fasst die Optionstypen zusammen.

%     Maßeinheit & \meta{Dim} & |listarraysep=0.5cm| \\\midrule

% \begin{itemize}

%   |twoupfootskip=|\meta{Dim}\quad(|10mm|)

% \end{decl}
